% Color definitions
\usepackage{color}

% Main font
\usepackage{fontspec}
\setmainfont[Path=fonts/,
    UprightFont=*-Regular.otf,
    ItalicFont=*-Italic.otf,
    BoldFont=*-Bold.otf,
    Ligatures={}]{Alegreya}


% Heading font
\usepackage{titlesec}
\newfontfamily\headingfont[Path=fonts/,
    UprightFont = *-Normal.otf,
    BoldFont = *-SemiBold.otf]{UGentPannoText}
\setsansfont[Path=fonts/,
    UprightFont = *-Normal.otf,
    BoldFont = *-SemiBold.otf]{UGentPannoText}
\titleformat{\chapter}[display]
  {\fontsize{16}{16}\headingfont}
  {\MakeUppercase\chaptertitlename\ \thechapter}{0pt}
  {\fontsize{28}{22}\selectfont\MakeUppercase}
\titleformat{\section}
  {\LARGE\headingfont}
  {\hspace{-2\smallgridsize}\makebox[1.5\smallgridsize][r]{\thesection}\hspace{.5\smallgridsize}}{0pt}{}
\titleformat{\subsection}{\Large\headingfont}
  {\hspace{-2\smallgridsize}\makebox[1.5\smallgridsize][r]{\thesubsection}\hspace{.5\smallgridsize}}{0pt}{}
\titleformat{\subsubsection}{\large\headingfont}
  {\hspace{-2\smallgridsize}\makebox[1.5\smallgridsize][r]{\thesubsubsection}\hspace{.5\smallgridsize}}{0pt}{}
\titleformat{\paragraph}
    {\headingfont\normalsize\bfseries}{\theparagraph}{1em}{}
\titleformat*{\subparagraph}{\bfseries\headingfont}
\titlespacing*{\paragraph}{0pt}{2.8ex plus 0.8ex minus .4ex}{-\parskip}

% Paragraph formatting
\usepackage{setspace}

\setlength{\parskip}{1.4ex}
\setlength{\parindent}{0pt}
\usepackage[inline]{enumitem}
\setlist{nosep, labelwidth=\parindent}
\setlist[1]{leftmargin=-\parindent}

% Precise typography
\usepackage{microtype}

\newcolumntype{R}[1]{>{\raggedleft\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}

% Increase vertical spacing in tabular
\renewcommand{\arraystretch}{1.4}

% Increase spacing between columns = spacing around floating figure
\setlength\columnsep{18pt}

\tikzset{
  font=\headingfont,
  every picture/.style={/utils/exec={\headingfont}},
  base-block/.append style = {
    font=\headingfont,
  }
}

\pgfplotsset{
  tick label style = {font=\headingfont, black!60},
  every axis label = {font=\headingfont, black},
  legend style = {
    font=\headingfont\small, 
  },
  label style/.append style = {font=\headingfont}
}
