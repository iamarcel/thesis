\newacronym{rnn}{RNN}{Recurrent Neural Network}

\newglossaryentry{motion-primitive}{
  name=motion primitive,
  description={A fundamental unit of motion or, in this context, of gesture. For
    example, someone might perform the motion primitive of holding both hands
    together and the moving them outwards to the sides while speaking the words
    ``bigger than.''}
}

\newglossaryentry{thought-vector}{
  name=thought vector,
  description={The name used to refer to the hidden state in an encoder-decoder
    architecture, often seen as a summary of the input sequence.}
}

\newglossaryentry{automaton}{
  name=automaton,
  description={A mechanized puppet. Atomatons can act and move autonomously but
    do not have the ability to interact with the outside world. Prime examples
    are those used in Disney theme parks.}
}

\newglossaryentry{iconic-gesture}{
  name=iconic gesture,
  description={A gesture that literally depicts an object or action that is
  being described. For example, ``walking'' with the fingers.}
}

\newglossaryentry{metaphoric-gesture}{
  name=metaphoric gesture,
  description={A gesture that indirectly represents an object or action. For
  example, making a motion of closing a lock in front of the lips.}
}

\newglossaryentry{deictic-gesture}{
  name=deictic gesture,
  description={A gesture that represents a position in space. The prototypical
  example is pointing at an object.}
}

\newglossaryentry{beat-gesture}{
  name=beat gesture,
  description={A gesture that on its own has little meaning, but is used in the
  context of a person speaking to, for example, clarify the parts of a logical
  progression or emphasize certain words.}
}
